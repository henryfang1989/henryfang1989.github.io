Solution #1

Use BFS to find all nodes in a island and mark those nodes to 0

Traverse through all nodes in grid, count the number of times that hit the 1

Because once we hit a node in island, the while island will be mark as 0. Every time, we hit a 1, which means we hit a new island.

Note: this approach modifies the grid

time complexity: O(m*n) space complexity: O(m*n)

Code

class Solution(object):
    def numIslands(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: int
        """
        # check valid grid
        if not grid or not grid[0]:
            return 0
        
        m, n = len(grid), len(grid[0])
        
        # find next valid position
        def next_positions(i, j):
            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                x, y = i+dx, j+dy
                if 0 <= x < m and 0 <= y < n and grid[x][y] == "1":
                    yield x, y
        # mark all nodes in a island to '0'
        def bfs(i, j):
            q = collections.deque([(i, j)])
            grid[i][j] = '0'
            while q:
                for x, y in next_positions(*q.popleft()):
                    grid[x][y] = '0'
                    q.append((x, y))
        # find number of islands
        res = 0
        for i in xrange(m):
            for j in xrange(n):
                if grid[i][j] == '1':
                    bfs(i, j)
                    res += 1
        return res

Solution #2

This approach uses the same similar idea as solution 1. But it does not modify the grid. It creates a visited matrix and use it to prevent the loop and determine the different island.

time complexity: O(m*n) space complexity: O(m*n)

Code

class Solution(object):
    def numIslands(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: int
        """
        if not grid or not grid[0]:
            return 0
        
        m, n = len(grid), len(grid[0])
        
        # use visited matrix to prevent loop and determine the different island
        visited = [[False for j in xrange(n)] for i in xrange(m)]
        
        # next valid position
        def next(i, j):
            for dx, dy in [(0,1), (0,-1), (1,0), (-1,0)]:
                x, y = i+dx, j+dy
                if 0 <= x < m and 0 <= y < n and grid[x][y] == '1' and not visited[x][y]:
                    yield x, y
        
        # find and visit all nodes in an island
        def bfs(i, j):
            q = collections.deque([(i, j)])
            visited[i][j] = True
            while q:
                for x, y in next(*q.popleft()):
                    visited[x][y] = True
                    q.append((x, y))
        
        # count the number of islands
        res = 0
        for i in xrange(m):
            for j in xrange(n):
                if grid[i][j] == '1' and not visited[i][j]:
                    bfs(i, j)
                    res += 1
        return res
