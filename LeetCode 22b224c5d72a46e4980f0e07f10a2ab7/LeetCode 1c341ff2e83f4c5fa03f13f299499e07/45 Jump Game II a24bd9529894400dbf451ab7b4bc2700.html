class Solution(object):
    def jump(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        # helper function to return the
        # the min steps that reaches to the end
        def helper(start, dp):
            if start >= len(nums)-1:
                return 0
            if dp[start] != -1:
                return dp[start]
            # if current num is 0, means cannot reach to the end
            if nums[start] == 0:
                dp[start] = float('inf')
            else:
                # try with different steps and return the min steps
                dp[start] = min([helper(start+i, dp) for i in xrange(1, nums[start]+1)]) + 1
            return dp[start]
        # dp used for skip the duplicated calculation
        dp = [-1]*len(nums)
        return helper(0, dp)

class Solution(object):
    def jump(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        # nums is None or len(nums) < 2, we don't have to jump
        # since we are already in index 0
        if not nums or len(nums) < 2:
            return 0
        # cur: the furthest position for current jump
        # next: the furthest position for next jump
        next = cur = nums[0]
        # count number of jumps
        jumps = 0
        # iterate each position. use the cur to compare
        # with current position to ensure we can reach here.
        # if current jump cannot reach here, we need make one more jump
        # to get further reachable position from next jump
        for i in xrange(0, len(nums)):
            if cur < i:
                jumps += 1
                cur = next
            # we need to find furthest position within
            # all of our reachable positions(gready strategy)
            next = max(next, i+nums[i])
        # finally we have to add 1 more jump in answer
        # because we only compare cur with current position
        # to know if we can reachable, but not really jump.
        return jumps+1

class Solution(object):
    def jump(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        # nums is None or len(nums) < 2, we don't have to jump
        # since we are already in index 0
        if not nums or len(nums) < 2:
            return 0
        # cur: the furthest position for current jump
        # next: the furthest position for next jump

        # with current position to ensure we can reach here.
        # if current jump cannot reach here, we need make one more jump
        # to get further reachable position from next jump

            if cur < i:
                jumps += 1
                # if furthest position in next jump
                # cannot reach to current position
                # it means we never be able to reach
                # to the end position
                if next < i:

            # all of our reachable positions(gready strategy)
