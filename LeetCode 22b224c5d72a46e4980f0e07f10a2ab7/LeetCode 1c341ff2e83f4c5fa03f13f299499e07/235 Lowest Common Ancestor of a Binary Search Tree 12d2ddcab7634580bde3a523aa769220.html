Solution #1

It's a BST, so the lowest common ancestor should be the node, whose value is between p.val and q.val . Use DFS to find the right node based on the current node value.

time complexity: O(logn) space complexity: O(logn)

Code

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """
        if p.val > q.val:
            p, q = q, p
        if not root:
            return None
        if p.val <= root.val <= q.val:
            return root
        if q.val < root.val:
            return self.lowestCommonAncestor(root.left, p, q)
        return self.lowestCommonAncestor(root.right, p, q)

Solution #2

Instead of using DFS, we could use iteration to do it. Use the stack to traverse the tree and compare the value if it is in between [p.val, q.val]

time complexity: O(n) space complexity: O(n)

Code

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """
        if p.val > q.val:
            p, q = q, p
        stack = [root]
        
        while stack:
            node = stack.pop()
            if p.val <= node.val <= q.val:
                return node
            if node.left:
                stack.append(node.left)
            if node.right:
                stack.append(node.right)
        return None

Solution #3

Use the same approach as solution #2. However, when we access the child tree, we compare the p and q with current node and filter out one of child tree

time complexity: O(logn) space complexity: O(logn)

Code

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """
        a, b = min(p.val, q.val), max(p.val, q.val)
        stack = [root]
        
        while stack:
            node = stack.pop()
            if a <= node.val <= b:
                return node
            if node.left and node.val > b:
                stack.append(node.left)
            elif node.right and node.val < a:
                stack.append(node.right)
        return None
